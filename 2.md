```mermaid
flowchart TD
    subgraph Administrador [Administrador / Actor]
        Inicio((Inicio)) --> AccederPanel[Accede al Panel de Administración]
        AccederPanel --> RellenarDatos[Rellena los datos del Recurso Educativo]
        RellenarDatos --> ClickCrear[Pulsa 'Crear Recurso']
    end

    subgraph Presentacion [Capa Presentación]
        ClickCrear --> RecibirPeticion[El Controlador recibe los datos de la Vista]
        RecibirPeticion --> TransferirSA[Pasa el DTO / Transfer al SA]
        MostrarExito[VistaPanelAdministrador muestra éxito] --> Fin(((Fin)))
        MostrarError[Vista muestra error al Admin] --> RellenarDatos
    end

    subgraph Negocio [Capa de Negocio - SA]
        TransferirSA --> ValidarReglas[SARecursoEducativo valida reglas de negocio]
        ValidarReglas --> CheckValido{¿Datos correctos?}
        CheckValido -- No --> RetornarError[Retorna excepción/error a Presentación]
        RetornarError --> MostrarError
        CheckValido -- Sí --> LlamarDAO[El SA llama a FactoriaDAO y pasa el transfer al DAO]
    end

    subgraph Integracion [Capa de Integración - DAO]
        LlamarDAO --> ExecuteQuery[DaoAdministrador/DaoRecurso ejecuta INSERT]
        ExecuteQuery --> UpdateBD[(Base de Datos)]
        UpdateBD --> CheckBD{¿Éxito en BD?}
        CheckBD -- Fallo SQL --> RetornarError
        CheckBD -- Éxito --> RetornarExito[Retorna el ID generado / Confirmación al SA]
        RetornarExito --> MostrarExito
    end

    %% Estilos
    classDef startend fill:#333,stroke:#333,stroke-width:2px,color:#fff;
    classDef bd fill:#f9f,stroke:#333,stroke-width:2px;
    class Inicio,Fin startend;
    class UpdateBD bd;
